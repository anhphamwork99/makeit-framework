<objective>
Record lessons learned after feature completion — structured event description, analysis, and actionable items for future improvement.
</objective>

<execution_context>
@.agent/skills/makeit-dev-be/be-support/lesson-learned.md
</execution_context>

<process>
  <step name="describe">
    Context, what happened, impact on timeline/quality.
  </step>

  <step name="analyze">
    What went well ✅, what didn't ❌, root cause.
  </step>

  <step name="action_items">
    Specific, actionable items with owners and deadlines.
  </step>

  <step name="save">
    Save entry, display summary, suggest sharing in retrospective.
  </step>
</process>

<success_criteria>
- [ ] Event described with context
- [ ] Analysis completed
- [ ] Action items defined
</success_criteria>
